-- Smooth ShiftLock Module
local SmoothShiftLock = nil
local shiftlockLoaded = false
local shiftlockWasEnabled = false

local function loadSmoothShiftLock()
    local success, module = pcall(function()
        return require(ReplicatedStorage:FindFirstChild("Systems"):FindFirstChild("Player"):FindFirstChild("Game"):FindFirstChild("SmoothShiftLock"))
    end)
    if not success then
        success, module = pcall(function()
            return require(ReplicatedStorage.Systems.Player.Game.SmoothShiftLock)
        end)
    end
    if success and module then
        SmoothShiftLock = module
        shiftlockLoaded = true
    end
end
loadSmoothShiftLock()

local function isShiftLockEnabled()
    if shiftlockLoaded and SmoothShiftLock then
        return SmoothShiftLock.Enabled
    else
        return UserInputService.MouseBehavior == Enum.MouseBehavior.LockCenter
    end
end

function setShiftLock(enabled)
    if enabled then
        shiftlockWasEnabled = isShiftLockEnabled()
        if not shiftlockWasEnabled then
            if shiftlockLoaded and SmoothShiftLock and not SmoothShiftLock.Enabled then
                pcall(function() SmoothShiftLock:ToggleShiftLock() end)
            else
                pcall(function()
                    local shiftlockKeybindEnum = lp.PlayerData.Settings.Keybinds.ShiftLock.Value
                    local shiftlockKeyCode = Enum.KeyCode[shiftlockKeybindEnum]
                    if shiftlockKeyCode then
                        VirtualInputManager:SendKeyEvent(true, shiftlockKeyCode, false, game)
                        VirtualInputManager:SendKeyEvent(false, shiftlockKeyCode, false, game)
                    end
                end)
            end
        end
    else
        if isShiftLockEnabled() then
            if shiftlockLoaded and SmoothShiftLock and SmoothShiftLock.Enabled then
                pcall(function() SmoothShiftLock:ToggleShiftLock() end)
            else
                pcall(function()
                    local shiftlockKeybindEnum = lp.PlayerData.Settings.Keybinds.ShiftLock.Value
                    local shiftlockKeyCode = Enum.KeyCode[shiftlockKeybindEnum]
                    if shiftlockKeyCode then
                        VirtualInputManager:SendKeyEvent(true, shiftlockKeyCode, false, game)
                        VirtualInputManager:SendKeyEvent(false, shiftlockKeyCode, false, game)
                    end
                end)
            end
        end
    end
end

local function shouldBlock()
    if blockChance >= 100 then return true end
    if blockChance <= 0 then return false end
    return math.random(1, 100) <= blockChance
end

local function isFacing(localRoot, targetRoot)
    if not facingCheckEnabled then return true end
    local dx = localRoot.Position.X - targetRoot.Position.X
    local dy = localRoot.Position.Y - targetRoot.Position.Y
    local dz = localRoot.Position.Z - targetRoot.Position.Z
    local mag = math.sqrt(dx*dx + dy*dy + dz*dz)
    if mag == 0 then return true end
    local invMag = 1 / mag
    local ux, uy, uz = dx * invMag, dy * invMag, dz * invMag
    local lv = targetRoot.CFrame.LookVector
    local dot = lv.X * ux + lv.Y * uy + lv.Z * uz
    return dot > customFacingDot
end

local function checkAndSwitchBlockType()
    if not autoSwitchBlockCharge then return end
    local currentTime = tick()
    if currentTime - lastBlockCheckTime < BLOCK_CHECK_INTERVAL then return end
    lastBlockCheckTime = currentTime
    refreshUIRefs()
    if not cachedBlockBtn then return end
    local blockCooldown = cachedBlockBtn:FindFirstChild("CooldownTime")
    if not blockCooldown then return end
    if blockCooldown.Text ~= "" and blockCooldown.Text ~= "0" then
        if autoblocktype ~= "Charge" then
            originalBlockType = autoblocktype
            autoblocktype = "Charge"
        end
    else
        if autoblocktype == "Charge" and originalBlockType ~= "Charge" then
            autoblocktype = originalBlockType
        end
    end
end

-- ===== PREDICTION FUNCTIONS =====
local function updatePositionHistory(character, position)
    if not character or not position then return end
    if not positionHistory[character] then positionHistory[character] = {} end
    local history = positionHistory[character]
    table.insert(history, position)
    if #history > HISTORY_SIZE then table.remove(history, 1) end
end

local function getMovementDirectionFromHistory(character)
    if not character or not positionHistory[character] then return Vector3.new(0,0,0) end
    local history = positionHistory[character]
    if #history < 2 then return Vector3.new(0,0,0) end
    local oldestPos = history[1]
    local newestPos = history[#history]
    local movement = newestPos - oldestPos
    local magnitude = movement.Magnitude
    local recentMovement = Vector3.new(0,0,0)
    if #history >= 3 then recentMovement = history[#history] - history[#history-1] end
    if magnitude > MOVE_THRESHOLD then return movement.Unit
    elseif recentMovement.Magnitude > MOVE_THRESHOLD then return recentMovement.Unit end
    return Vector3.new(0,0,0)
end

local function getInstantaneousDirection(character, currentPos)
    if not character or not currentPos then return Vector3.new(0,0,0) end
    if not positionHistory[character] or #positionHistory[character] < 2 then return Vector3.new(0,0,0) end
    local history = positionHistory[character]
    local lastPos = history[#history]
    local movement = currentPos - lastPos
    if movement.Magnitude > MOVE_THRESHOLD then return movement.Unit end
    return Vector3.new(0,0,0)
end

local function getPredictedPosition(character, predictionMultiplier)
    if not character then return nil end
    local root = character:FindFirstChild("HumanoidRootPart") or character:FindFirstChild("Head") or character:FindFirstChild("Torso")
    if not root then return nil end
    local currentPos = root.Position
    updatePositionHistory(character, currentPos)
    local moveDir = getMovementDirectionFromHistory(character)
    local instantDir = getInstantaneousDirection(character, currentPos)
    local finalDir = instantDir
    if instantDir.Magnitude <= MOVE_THRESHOLD then finalDir = moveDir end
    local currentPredictionDistance = BASE_PREDICTION_DISTANCE * (predictionMultiplier or aimbotPredictionMultiplier)
    if finalDir.Magnitude > MOVE_THRESHOLD then return currentPos + (finalDir * currentPredictionDistance) else return currentPos end
end

local function createTargetSphere(targetCharacter, sphereColor)
    if currentTargetSphere then pcall(function() currentTargetSphere:Destroy() end); currentTargetSphere = nil end
    if currentSphereConnection then pcall(function() currentSphereConnection:Disconnect() end); currentSphereConnection = nil end
    if not targetCharacter then return nil end
    local root = targetCharacter:FindFirstChild("HumanoidRootPart") or targetCharacter:FindFirstChild("Head") or targetCharacter:FindFirstChild("Torso")
    if not root then return nil end
    local sphere = Instance.new("Part")
    sphere.Shape = Enum.PartType.Ball
    sphere.Size = Vector3.new(2,2,2)
    sphere.Material = Enum.Material.Neon
    sphere.Color = sphereColor or Color3.fromRGB(255,0,0)
    sphere.Anchored = true
    sphere.CanCollide = false
    sphere.Name = "TargetSphere"
    sphere.Parent = workspace
    sphere.Position = root.Position
    updatePositionHistory(targetCharacter, root.Position)
    local connection
    connection = RunService.Heartbeat:Connect(function()
        if not targetCharacter or not targetCharacter.Parent or not sphere.Parent then
            sphere:Destroy(); if connection then connection:Disconnect() end; currentTargetSphere = nil; currentSphereConnection = nil; return
        end
        local humanoid = targetCharacter:FindFirstChildOfClass("Humanoid")
        if not humanoid or humanoid.Health <= 0 then
            sphere:Destroy(); if connection then connection:Disconnect() end; currentTargetSphere = nil; currentSphereConnection = nil; return
        end
        root = targetCharacter:FindFirstChild("HumanoidRootPart") or targetCharacter:FindFirstChild("Head") or targetCharacter:FindFirstChild("Torso")
        if not root then
            sphere:Destroy(); if connection then connection:Disconnect() end; currentTargetSphere = nil; currentSphereConnection = nil; return
        end
        local currentPos = root.Position
        updatePositionHistory(targetCharacter, currentPos)
        local moveDir = getMovementDirectionFromHistory(targetCharacter)
        local instantDir = getInstantaneousDirection(targetCharacter, currentPos)
        local finalDir = instantDir
        if instantDir.Magnitude <= MOVE_THRESHOLD then finalDir = moveDir end
        local currentPredictionDistance = BASE_PREDICTION_DISTANCE * aimbotPredictionMultiplier
        local targetPosition = finalDir.Magnitude > MOVE_THRESHOLD and (currentPos + (finalDir * currentPredictionDistance)) or currentPos
        sphere.Position = sphere.Position:Lerp(targetPosition, SPHERE_SMOOTHNESS)
    end)
    currentTargetSphere = sphere
    currentSphereConnection = connection
    return sphere, connection
end

-- ===== GET VALID TARGETS FOR AIMBOT =====
local function getValidTargets()
    local targets = {}
    local myChar = lp.Character
    if not myChar then return targets end
    local myRoot = myChar:FindFirstChild("HumanoidRootPart")
    if not myRoot then return targets end
    local myTeam, myCharacter = getMyCharacterInfo()
    local isElliot = (myTeam == "Survivor" and myCharacter == "Elliot")
    local viewportSize = Camera.ViewportSize
    local screenCenter = Vector2.new(viewportSize.X/2, viewportSize.Y/2)
    local function processTarget(obj)
        local usernameAttr = obj:GetAttribute("Username")
        if usernameAttr and usernameAttr == lp.Name then return end
        local humanoid = obj:FindFirstChildOfClass("Humanoid")
        if not humanoid or humanoid.Health <= 0 then return end
        local targetPart = obj:FindFirstChild("HumanoidRootPart") or obj:FindFirstChild("Head") or obj:FindFirstChild("Torso")
        if not targetPart then return end
        local distance = (targetPart.Position - myRoot.Position).Magnitude
        if distance > aimbotMaxDistance then return end
        if aimbotUseFOV then
            local screenPos, onScreen = Camera:WorldToViewportPoint(targetPart.Position)
            if onScreen then
                local screenVector = Vector2.new(screenPos.X, screenPos.Y)
                local distanceFromCenter = (screenVector - screenCenter).Magnitude
                if distanceFromCenter <= aimbotFOV then
                    table.insert(targets, { obj = obj, part = targetPart, distance = distance, screenDistance = distanceFromCenter })
                end
            end
        else
            table.insert(targets, { obj = obj, part = targetPart, distance = distance, screenDistance = 0 })
        end
    end
    if isElliot then
        for _, survivor in ipairs(SurvivorsFolder:GetChildren()) do processTarget(survivor) end
    elseif myTeam == "Killer" then
        for _, survivor in ipairs(SurvivorsFolder:GetChildren()) do processTarget(survivor) end
    elseif myTeam == "Survivor" then
        for _, killer in ipairs(KillersFolder:GetChildren()) do processTarget(killer) end
    end
    if aimbotUseFOV then
        table.sort(targets, function(a,b) return a.screenDistance < b.screenDistance end)
    else
        table.sort(targets, function(a,b) return a.distance < b.distance end)
    end
    return targets
end

local function getNearestKiller()
    local myChar = lp.Character
    if not myChar then return nil end
    local myRoot = myChar:FindFirstChild("HumanoidRootPart")
    if not myRoot then return nil end
    local nearestKiller = nil
    local nearestDistance = math.huge
    for _, killer in ipairs(KillersFolder:GetChildren()) do
        local usernameAttr = killer:GetAttribute("Username")
        if not (usernameAttr and usernameAttr == lp.Name) then
            local targetHrp = killer:FindFirstChild("HumanoidRootPart")
            if targetHrp then
                local humanoid = killer:FindFirstChildOfClass("Humanoid")
                if humanoid and humanoid.Health > 0 then
                    local distance = (targetHrp.Position - myRoot.Position).Magnitude
                    if distance < nearestDistance then
                        nearestDistance = distance
                        nearestKiller = killer
                    end
                end
            end
        end
    end
    return nearestKiller
end

function startGeneralAimbot(duration, customSmoothness)
    if not aimbotEnabled then return end
    if aimbotActive then return end
    local targets = getValidTargets()
    if #targets == 0 then return end
    local target = targets[1]
    aimbotCurrentTarget = target.obj
    aimbotActive = true
    aimbotTargetCount = aimbotTargetCount + 1
    setShiftLock(true)
    local originalSmoothness = aimbotSmoothness
    if customSmoothness then aimbotSmoothness = customSmoothness end
    createTargetSphere(aimbotCurrentTarget, Color3.fromRGB(255,0,0))
    local cameraConnection
    cameraConnection = RunService.RenderStepped:Connect(function()
        if not aimbotActive or not aimbotCurrentTarget then if cameraConnection then cameraConnection:Disconnect() end; return end
        local myChar = lp.Character
        if not myChar then return end
        local humanoid = aimbotCurrentTarget:FindFirstChildOfClass("Humanoid")
        if not humanoid or humanoid.Health <= 0 then stopAimbot(); if cameraConnection then cameraConnection:Disconnect() end; return end
        local targetPos = getPredictedPosition(aimbotCurrentTarget, aimbotPredictionMultiplier)
        if not targetPos then stopAimbot(); if cameraConnection then cameraConnection:Disconnect() end; return end
        local camera = Camera
        local cameraPos = camera.CFrame.Position
        local targetCFrame = CFrame.lookAt(cameraPos, targetPos)
        local currentCFrame = camera.CFrame
        local currentLook = currentCFrame.LookVector
        local targetLook = targetCFrame.LookVector
        local smoothLook = currentLook:Lerp(targetLook, aimbotSmoothness)
        local newCFrame = CFrame.new(cameraPos, cameraPos + smoothLook)
        camera.CFrame = newCFrame
    end)
    if aimbotActiveTimer then pcall(function() task.cancel(aimbotActiveTimer) end) end
    aimbotActiveTimer = task.delay(duration, function()
        if customSmoothness then aimbotSmoothness = originalSmoothness end
        aimbotActive = false; aimbotCurrentTarget = nil
        if cameraConnection then cameraConnection:Disconnect() end
        if currentTargetSphere then pcall(function() currentTargetSphere:Destroy() end); currentTargetSphere = nil end
        setShiftLock(false)
    end)
end

-- ===== AUTO BLOCK FUNCTIONS =====
local function checkForAttackSounds()
    if not autoBlockForGuest then return end
    local now = tick()
    if now - lastBlockTime < BLOCK_COOLDOWN then return end
    local myChar = lp.Character
    local myRoot = myChar and myChar:FindFirstChild("HumanoidRootPart")
    if not myRoot then return end
    if not KillersFolder then return end
    local isGuest1337, guestChar = amIGuest1337()
    if not isGuest1337 then return end
    for _, killer in ipairs(KillersFolder:GetChildren()) do
        local usernameAttr = killer:GetAttribute("Username")
        if not (usernameAttr and usernameAttr == lp.Name) then
            local killerHrp = killer:FindFirstChild("HumanoidRootPart")
            if killerHrp then
                local distance = (killerHrp.Position - myRoot.Position).Magnitude
                if distance <= detectionRange then
                    if (not facingCheckEnabled) or isFacing(myRoot, killerHrp) then
                        if shouldBlock() then
                            local shouldBlockThisAttack = false
                            if killer.Name == "Slasher" then
                                local humanoid = killer:FindFirstChildOfClass("Humanoid")
                                if humanoid then
                                    local animator = humanoid:FindFirstChildOfClass("Animator")
                                    if animator then
                                        for _, track in ipairs(animator:GetPlayingAnimationTracks()) do
                                            local animId = extractNumericAnimationId(track.Animation.AnimationId)
                                            if animId and autoBlockTriggerAnimations[animId] then
                                                shouldBlockThisAttack = true
                                                break
                                            end
                                        end
                                    end
                                end
                            end
                            if not shouldBlockThisAttack then
                                for _, sound in ipairs(killer:GetDescendants()) do
                                    if sound:IsA("Sound") and sound.IsPlaying then
                                        local soundId = extractNumericSoundId(sound)
                                        if soundId and autoBlockTriggerSounds[soundId] then
                                            shouldBlockThisAttack = true
                                            break
                                        end
                                    end
                                end
                            end
                            if shouldBlockThisAttack then
                                if blockdelay > 0 then task.wait(blockdelay) end
                                refreshUIRefs()
                                if isHelpless() then break end
                                if autoblocktype == "Block" then
                                    if cachedCooldown and cachedCooldown.Text == "" then
                                        pressKey(Enum.KeyCode.Q)
                                        lastBlockTime = now
                                        return
                                    end
                                elseif autoblocktype == "Charge" then
                                    if cachedChargeBtn and cachedChargeBtn:FindFirstChild("CooldownTime") and cachedChargeBtn.CooldownTime.Text == "" then
                                        pressKey(Enum.KeyCode.Q)
                                        lastBlockTime = now
                                        return
                                    end
                                end
                            end
                        end
                    end
                end
            end
        end
    end
end

local function checkSpecialKillerStatus()
    local character = lp.Character
    if not character then
        if isPlayingAsSpecialKiller then
            isPlayingAsSpecialKiller = false
            currentKillerType = nil
            facingCheckEnabled = originalFacingCheckState
        end
        return false
    end
    local allSlashers = getAllCharactersByName("Slasher")
    local allJohnDoes = getAllCharactersByName("JohnDoe")
    local allOnexOnexOnexOnes = getAllCharactersByName("1x1x1x1")
    for _, slasher in ipairs(allSlashers) do
        local usernameAttr = slasher:GetAttribute("Username")
        if usernameAttr and usernameAttr == lp.Name then
            if not isPlayingAsSpecialKiller or currentKillerType ~= "Slasher" then
                isPlayingAsSpecialKiller = true
                currentKillerType = "Slasher"
                originalFacingCheckState = facingCheckEnabled
                facingCheckEnabled = false
            end
            return true
        end
    end
    for _, johnDoe in ipairs(allJohnDoes) do
        local usernameAttr = johnDoe:GetAttribute("Username")
        if usernameAttr and usernameAttr == lp.Name then
            if not isPlayingAsSpecialKiller or currentKillerType ~= "JohnDoe" then
                isPlayingAsSpecialKiller = true
                currentKillerType = "JohnDoe"
                originalFacingCheckState = facingCheckEnabled
                facingCheckEnabled = false
            end
            return true
        end
    end
    for _, onexone in ipairs(allOnexOnexOnexOnes) do
        local usernameAttr = onexone:GetAttribute("Username")
        if usernameAttr and usernameAttr == lp.Name then
            if not isPlayingAsSpecialKiller or currentKillerType ~= "1x1x1x1" then
                isPlayingAsSpecialKiller = true
                currentKillerType = "1x1x1x1"
                originalFacingCheckState = facingCheckEnabled
                facingCheckEnabled = false
            end
            return true
        end
    end
    if isPlayingAsSpecialKiller then
        isPlayingAsSpecialKiller = false
        currentKillerType = nil
        facingCheckEnabled = originalFacingCheckState
    end
    return false
end

local function safeDestroyCircle(circle)
    if not circle then return end
    pcall(function() if circle.Parent then circle:Destroy() end end)
end

    for _, circle in pairs(survivorCircles) do safeDestroyCircle(circle) end
    survivorCircles = {}

function startStunImmuneLoop()
    if stunImmuneLoop then task.cancel(stunImmuneLoop); stunImmuneLoop = nil end

    stunImmuneLoop = task.spawn(function()
        while stunImmuneEnabled do
            task.wait(0.5)
            scanAllKillersForStunImmune()
        end
    end)
end

            if isGuest1337 then startGeneralAimbot(0.95) end
        end
    end)

-- ===== SURVIVOR ABILITY COOLDOWN TRACKER FUNCTIONS =====
local function formatCooldownTime(remainingSeconds)
    if remainingSeconds <= 0 then return "0.0"
    elseif remainingSeconds < DECIMAL_START then return string.format("%.1f", remainingSeconds)
    else return tostring(math.floor(remainingSeconds)) end
end

local function getAbilityName(survivorName)
    if survivorName == "Shedletsky" then return "Sword"
    elseif survivorName == "TwoTime" then return "Dagger"
    elseif survivorName == "Chance" then return "Gun"
    elseif survivorName == "JaneDoe" then return "Hatchet"
    elseif survivorName == "Guest1337" then return "Block"
    elseif survivorName == "Elliot" then return "Pizza"
    elseif survivorName == "007n7" or survivorName == "7n7" then return "Clone"
    end
    return "Ability"
end

local function getCooldown(survivorName)
    if survivorName == "Shedletsky" then return SHEDLETSKY_COOLDOWN
    elseif survivorName == "TwoTime" then return TWOTIME_COOLDOWN
    elseif survivorName == "Chance" then return CHANCE_COOLDOWN
    elseif survivorName == "JaneDoe" then return JANEDOE_COOLDOWN
    elseif survivorName == "Guest1337" then return GUEST1337_COOLDOWN
    elseif survivorName == "Elliot" then return ELLIOT_COOLDOWN
    elseif survivorName == "007n7" or survivorName == "7n7" then return SEVENN7_COOLDOWN
    end
    return 40
end

local function createSurvivorCooldownDisplay(survivor, cooldownEndTime, abilityName)
    if activeCooldownTimers[survivor] then
        local existing = activeCooldownTimers[survivor]
        if existing.billboard then pcall(function() existing.billboard:Destroy() end) end
        if existing.connection then existing.connection:Disconnect() end
        activeCooldownTimers[survivor] = nil
    end
    local hrp = survivor:FindFirstChild("HumanoidRootPart") or survivor:FindFirstChild("Head") or survivor:FindFirstChild("Torso")
    if not hrp then return end
    local billboard = Instance.new("BillboardGui")
    billboard.Name = "AbilityCooldown"
    billboard.Size = UDim2.new(0,120,0,40)
    billboard.StudsOffset = Vector3.new(0,3,0)
    billboard.AlwaysOnTop = true
    billboard.Parent = hrp
    local textLabel = Instance.new("TextLabel")
    textLabel.Name = "CooldownText"
    textLabel.Size = UDim2.new(1,0,1,0)
    textLabel.BackgroundTransparency = 1
    textLabel.Text = abilityName .. ": " .. formatCooldownTime(getCooldown(survivor.Name))
    textLabel.TextStrokeTransparency = 0
    textLabel.TextStrokeColor3 = Color3.fromRGB(0,0,0)
    textLabel.TextScaled = true
    textLabel.Font = Enum.Font.GothamBold
    textLabel.Parent = billboard
    local timerData = { endTime = cooldownEndTime, billboard = billboard, textLabel = textLabel, abilityName = abilityName, survivorName = survivor.Name, connection = nil }
    local connection
    connection = RunService.Heartbeat:Connect(function()
        if not survivor or not survivor.Parent then
            if connection then connection:Disconnect() end
            if billboard then pcall(function() billboard:Destroy() end) end
            activeCooldownTimers[survivor] = nil
            return
        end
        local currentTime = tick()
        local remaining = timerData.endTime - currentTime
        if remaining <= 0.1 then
            if connection then connection:Disconnect() end
            if billboard then pcall(function() billboard:Destroy() end) end
            activeCooldownTimers[survivor] = nil
        else
            local formattedTime = formatCooldownTime(remaining)
            textLabel.Text = timerData.abilityName .. ": " .. formattedTime
            if remaining < 5 then textLabel.TextColor3 = Color3.fromRGB(255,0,0)
            elseif remaining < 10 then textLabel.TextColor3 = Color3.fromRGB(255,165,0)
            else textLabel.TextColor3 = Color3.fromRGB(255,255,255) end
        end
    end)
    timerData.connection = connection
    activeCooldownTimers[survivor] = timerData
end

local function listenForCooldownAbility(survivor)
    if not survivor then return end
    if isLocalPlayer(survivor) then return end
    if survivor:GetAttribute("CooldownListener") then return end
    survivor:SetAttribute("CooldownListener", true)
    local survivorName = survivor.Name
    local function onAbilityDetected(cooldown)
        local cooldownEndTime = tick() + cooldown
        local abilityName = getAbilityName(survivorName)
        createSurvivorCooldownDisplay(survivor, cooldownEndTime, abilityName)
    end
    local function onGuest1337BlockDetected()
        local now = tick()
        local lastTrigger = lastGuest1337Trigger[survivor] or 0
        if now - lastTrigger >= GUEST1337_ANTI_SPAM_TIME then
            lastGuest1337Trigger[survivor] = now
            onAbilityDetected(GUEST1337_COOLDOWN)
        end
    end
    local function onChanceAbilityDetected()
        local now = tick()
        local lastTrigger = lastChanceTrigger[survivor] or 0
        if now - lastTrigger >= CHANCE_ANTI_SPAM_TIME then
            lastChanceTrigger[survivor] = now
            onAbilityDetected(CHANCE_COOLDOWN)
        end
    end
    local function onSoundPlayed(sound)
        if not sound.IsPlaying then return end
        local soundId = extractNumericSoundId(sound)
        if survivorName == "Shedletsky" and soundId and shedletskySwingSounds[soundId] then onAbilityDetected(SHEDLETSKY_COOLDOWN)
        elseif survivorName == "TwoTime" and soundId and twotimeSounds[soundId] then onAbilityDetected(TWOTIME_COOLDOWN)
        elseif survivorName == "Chance" and soundId and chanceSounds[soundId] then onChanceAbilityDetected()
        elseif survivorName == "JaneDoe" and soundId and janeDoeSounds[soundId] then onAbilityDetected(JANEDOE_COOLDOWN) end
    end
    survivor.DescendantAdded:Connect(function(descendant)
        if descendant:IsA("Sound") then
            if descendant.IsPlaying then onSoundPlayed(descendant) end
            descendant.Played:Connect(function() onSoundPlayed(descendant) end)
        end
    end)
    local head = survivor:FindFirstChild("Head")
    if head then
        local lastTransparency = head.Transparency
        head:GetPropertyChangedSignal("Transparency"):Connect(function()
            if head.Transparency ~= lastTransparency then
                lastTransparency = head.Transparency
                if head.Transparency ~= 0 then onAbilityDetected(SEVENN7_COOLDOWN) end
            end
        end)
    end
    local humanoid = survivor:FindFirstChildOfClass("Humanoid")
    if humanoid then
        local function setupAnimator(animator)
            if not animator then return end
            animator.AnimationPlayed:Connect(function(animationTrack)
                local animId = extractNumericAnimationId(animationTrack.Animation.AnimationId)
                if survivorName == "Chance" and animId and chanceAnimIds[animId] then onChanceAbilityDetected()
                elseif survivorName == "Guest1337" and animId and guest1337AllAnims[animId] then onGuest1337BlockDetected()
                elseif survivorName == "Elliot" and animId and elliotThrowAnimation[animId] then onAbilityDetected(ELLIOT_COOLDOWN) end
            end)
        end
        local animator = humanoid:FindFirstChildOfClass("Animator")
        if animator then setupAnimator(animator) end
        humanoid.ChildAdded:Connect(function(child)
            if child:IsA("Animator") then setupAnimator(child) end
        end)
    end
end

            for _, killer in ipairs(killersFolder:GetChildren()) do

local function isSlasherEnraged(slasher)
    local head = slasher:FindFirstChild("Head")
    if head then
        local ragingPaceAttachment = head:FindFirstChild("RagingPaceAttachment")
        if ragingPaceAttachment then
            local ragingPace = ragingPaceAttachment:FindFirstChild("RagingPace")
            if ragingPace and ragingPace:IsA("ParticleEmitter") then
                return ragingPace.Enabled == true
            end
        end
    end
    if head then
        local ragingPace = head:FindFirstChild("RagingPace")
        if ragingPace then
            return ragingPace.Enabled == true
        end
    end
    local plate = slasher:FindFirstChild("Plate")
    if plate then
        local stringObj = plate:FindFirstChild("String")
        if stringObj then
            return stringObj.Transparency == 1
        end
    end
    return false
end

local function setupSlasherEnrageTracking(slasher)
    local head = slasher:FindFirstChild("Head")
    local function updateEnrageState()
        slasherIsEnraged[slasher] = isSlasherEnraged(slasher)
    end
    if head then
        local ragingPaceAttachment = head:FindFirstChild("RagingPaceAttachment")
        if ragingPaceAttachment then
            local ragingPace = ragingPaceAttachment:FindFirstChild("RagingPace")
            if ragingPace and ragingPace:IsA("ParticleEmitter") then
                ragingPace:GetPropertyChangedSignal("Enabled"):Connect(updateEnrageState)
            end
        end
    end
    if head then
        local ragingPace = head:FindFirstChild("RagingPace")
        if ragingPace then
            ragingPace:GetPropertyChangedSignal("Enabled"):Connect(updateEnrageState)
        end
    end
    local plate = slasher:FindFirstChild("Plate")
    if plate then
        local stringObj = plate:FindFirstChild("String")
        if stringObj then
            stringObj:GetPropertyChangedSignal("Transparency"):Connect(updateEnrageState)
        end
    end
    updateEnrageState()
end

local function listenForSlasherAbilities(slasher)
    if not slasher then return end
    if slasher:GetAttribute("SlasherListener") then return end
    slasher:SetAttribute("SlasherListener", true)
    setupSlasherEnrageTracking(slasher)
    local function onAbilityDetected(cooldown, abilityName)
        local cooldownEndTime = tick() + cooldown
        createSlasherCooldownDisplay(slasher, cooldownEndTime, abilityName)
    end
    local function getBeheadCooldown()
        if isSlasherEnraged(slasher) then return SLASHER_BEHEAD_ENRAGED else return SLASHER_BEHEAD_NORMAL end
    end
    local function getGashingWoundCooldown()
        if isSlasherEnraged(slasher) then return SLASHER_GASHING_WOUND_ENRAGED else return SLASHER_GASHING_WOUND_NORMAL end
    end
    local function onRagingPaceUsed()
        local cooldownEndTime = tick() + SLASHER_RAGING_PACE_COOLDOWN
        createSlasherCooldownDisplay(slasher, cooldownEndTime, "Raging Pace")
    end
    local humanoid = slasher:FindFirstChildOfClass("Humanoid")
    if humanoid then
        local function setupAnimator(animator)
            if not animator then return end
            animator.AnimationPlayed:Connect(function(animationTrack)
                local animId = extractNumericAnimationId(animationTrack.Animation.AnimationId)
                local isBehead = false
                for _, id in ipairs(BEHEAD_ANIMATIONS) do if animId == id then isBehead = true; break end end
                if isBehead then onAbilityDetected(getBeheadCooldown(), "Behead") end
                local isGashingWound = false
                for _, id in ipairs(GASHING_WOUND_ANIMATIONS) do if animId == id then isGashingWound = true; break end end
                if isGashingWound then onAbilityDetected(getGashingWoundCooldown(), "Gashing Wound") end
                local isRagingPaceStartup = false
                for _, id in ipairs(RAGING_PACE_STARTUP_ANIMATIONS) do if animId == id then isRagingPaceStartup = true; break end end
                if isRagingPaceStartup then onRagingPaceUsed() end
            end)
        end
        local animator = humanoid:FindFirstChildOfClass("Animator")
        if animator then setupAnimator(animator) end
        humanoid.ChildAdded:Connect(function(child)
            if child:IsA("Animator") then setupAnimator(child) end
        end)
    end
end

        if not isLocalPlayer(slasher) then listenForSlasherAbilities(slasher) end

local function reduceSixerCooldownByBloodOrb(sixer)

    for abilityName, timerData in pairs(sixerActiveCooldownTimers[sixer]) do
        if timerData and timerData.endTime then
            local currentRemaining = timerData.endTime - tick()
            local newRemaining = currentRemaining - BLOOD_ORB_REDUCTION
            if newRemaining <= 0 then
                if timerData.billboard then pcall(function() timerData.billboard:Destroy() end) end
                if timerData.connection then timerData.connection:Disconnect() end
                sixerActiveCooldownTimers[sixer][abilityName] = nil
            else
                timerData.endTime = tick() + newRemaining
                if timerData.textLabel then
                    timerData.textLabel.Text = abilityName .. ": " .. formatSixerCooldownTime(newRemaining)
                end
            end
        end
    end

local function clearPendingNoliVoidRush()
    if noliPendingTimer then
        task.cancel(noliPendingTimer)
        noliPendingTimer = nil
    end
    noliPendingNoli = nil
end

local function listenForNoliVoidRush(noli)
    local humanoid = noli:FindFirstChildOfClass("Humanoid")
    if not humanoid then return end
    
    local function setupAnimator(animator)
        if not animator then return end
        
        animator.AnimationPlayed:Connect(function(animationTrack)
            local animId = extractNumericAnimationId(animationTrack.Animation.AnimationId)
            
            for _, id in ipairs(VOID_RUSH_INSTANT_ANIMATIONS) do
                if animId == id then
                    local now = tick()
                    if now - noliLastVoidRushTrigger >= VOID_RUSH_ANTI_SPAM then
                        noliLastVoidRushTrigger = now
                        clearPendingNoliVoidRush()
                        local cooldownEndTime = tick() + VOID_RUSH_COOLDOWN
                        createNoliCooldownDisplay(noli, cooldownEndTime, "Void Rush")
                    end
                    return
                end
            end
            
            for _, startAnim in ipairs(VOID_RUSH_START_ANIMATIONS) do
                if animId == startAnim then
                    clearPendingNoliVoidRush()
                    noliPendingNoli = noli
                    noliPendingTimer = task.delay(VOID_RUSH_WINDOW, function()
                        local now = tick()
                        if now - noliLastVoidRushTrigger >= VOID_RUSH_ANTI_SPAM then
                            noliLastVoidRushTrigger = now
                            local cooldownEndTime = tick() + VOID_RUSH_COOLDOWN
                            createNoliCooldownDisplay(noli, cooldownEndTime, "Void Rush")
                        end
                        clearPendingNoliVoidRush()
                    end)
                    return
                end
            end
            
            for _, cancelAnim in ipairs(VOID_RUSH_CANCEL_ANIMATIONS) do
                if animId == cancelAnim then
                    if noliPendingNoli == noli then
                        clearPendingNoliVoidRush()
                    end
                    return
                end
            end
        end)
    end
    
    local animator = humanoid:FindFirstChildOfClass("Animator")
    if animator then
        setupAnimator(animator)
    end
    
    humanoid.ChildAdded:Connect(function(child)
        if child:IsA("Animator") then
            setupAnimator(child)
        end
    end)
end

    if noli:GetAttribute("NoliListener") then return end
    noli:SetAttribute("NoliListener", true)

local function formatJohnDoeCooldownTime(remainingSeconds)
    if remainingSeconds <= 0 then return "0.0"
    elseif remainingSeconds < 10 then return string.format("%.1f", remainingSeconds)
    else return tostring(math.floor(remainingSeconds)) end
end

local function getSurvivorClosestToCursor()
    local myChar = lp.Character
    if not myChar then return nil end
    local myRoot = myChar:FindFirstChild("HumanoidRootPart")
    if not myRoot then return nil end
    local mousePosition = UserInputService:GetMouseLocation()
    local validTargets = {}
    for _, survivor in ipairs(SurvivorsFolder:GetChildren()) do
        local usernameAttr = survivor:GetAttribute("Username")
        if not (usernameAttr and usernameAttr == lp.Name) then
            local humanoid = survivor:FindFirstChildOfClass("Humanoid")
            if humanoid and humanoid.Health > 0 then
                local targetHrp = survivor:FindFirstChild("HumanoidRootPart")
                if targetHrp then
                    local screenPos, onScreen = Camera:WorldToViewportPoint(targetHrp.Position)
                    if onScreen then
                        local screenVector = Vector2.new(screenPos.X, screenPos.Y)
                        local distanceToCursor = (screenVector - mousePosition).Magnitude
                        table.insert(validTargets, { survivor = survivor, screenDistance = distanceToCursor })
                    end
                end
            end
        end
    end
    table.sort(validTargets, function(a, b) return a.screenDistance < b.screenDistance end)
    if #validTargets > 0 then return validTargets[1].survivor end
    local nearestByDistance = nil
    local nearestDist = math.huge
    for _, survivor in ipairs(SurvivorsFolder:GetChildren()) do
        local usernameAttr = survivor:GetAttribute("Username")
        if not (usernameAttr and usernameAttr == lp.Name) then
            local humanoid = survivor:FindFirstChildOfClass("Humanoid")
            if humanoid and humanoid.Health > 0 then
                local targetHrp = survivor:FindFirstChild("HumanoidRootPart")
                if targetHrp then
                    local distance = (targetHrp.Position - myRoot.Position).Magnitude
                    if distance < nearestDist then
                        nearestDist = distance
                        nearestByDistance = survivor
                    end
                end
            end
        end
    end
    return nearestByDistance
end

function setupSilentAim()
    if isSilentAimHooked then 
        return true 
    end
    
    local module = nil
    local success, mod = pcall(function()
        return require(ReplicatedStorage:FindFirstChild("Systems"):FindFirstChild("Player"):FindFirstChild("Miscellaneous"):FindFirstChild("GetPlayerMousePosition"))
    end)
    if success and mod then
        module = mod
    else
        success, mod = pcall(function()
            return require(ReplicatedStorage.Systems.Player.Miscellaneous.GetPlayerMousePosition)
        end)
        if success and mod then
            module = mod
        end
    end
    
    if module then
        mousePositionModule = module
        originalGetMousePos = module.GetMousePos
        
        local function hookedGetMousePos(self, ...)
            if silentAimActive and currentTargetSphere then
                return currentTargetSphere.Position
            end
            return originalGetMousePos(self, ...)
        end
        
        module.GetMousePos = hookedGetMousePos
        isSilentAimHooked = true
        return true
    end
    return false
end

function restoreSilentAim()
    if mousePositionModule and originalGetMousePos then
        mousePositionModule.GetMousePos = originalGetMousePos
        isSilentAimHooked = false
    end
    mousePositionModule = nil

function useSilentAim()
    if not dusekkarEnabled then return false end
    if not amIDusekkar() then return false end
    
    if not isSilentAimHooked then
        setupSilentAim()
    end
    
    local target = nil
    if dusekkarTargetMode == "Survivors" then
        target = getSurvivorClosestToCursor()
        if target then
            SendNotif("Dusekkar", "Targeting survivor: " .. target.Name, 1.5)
        end
    else
        target = getNearestKillerForDusekkar()
        if target then
            SendNotif("Dusekkar", "Targeting killer: " .. target.Name, 1.5)
        end
    end
    
    if not target then 
        SendNotif("Dusekkar", "No target found!", 1)
        return false 
    end
    
    if isPlasmaBeamOnCooldown() then 
        SendNotif("Dusekkar", "Plasma Beam on cooldown!", 1)
        return false 
    end
    
    currentDusekkarTarget = target
    silentAimActive = true
    dusekkarActive = true
    
    createTargetSphere(target, Color3.fromRGB(0, 170, 255))
    
    local success = firePlasmaBeam()
    
    if success then
        if dusekkarTimer then task.cancel(dusekkarTimer) end
        dusekkarTimer = task.delay(2.5, function()
            stopDusekkar()
        end)
        return true
    else
        stopDusekkar()
        return false
    end
end

local function initCharacterListeners()
    if SurvivorsFolder then
        SurvivorsFolder.ChildAdded:Connect(function(newSurvivor)
            local root = newSurvivor:WaitForChild("HumanoidRootPart", 5)
            if root then updatePositionHistory(newSurvivor, root.Position) end
        end)
    end
    if KillersFolder then
        KillersFolder.ChildAdded:Connect(function(newKiller)
            local root = newKiller:WaitForChild("HumanoidRootPart", 5)
            if root then updatePositionHistory(newKiller, root.Position) end
        end)
    end
    local function watchCharacterRemoval(folder)
        if folder then
            folder.ChildRemoved:Connect(function(character)
                if positionHistory[character] then positionHistory[character] = nil end
                if aimbotCurrentTarget == character then stopAimbot() end
                if currentDusekkarTarget == character then stopDusekkar() end
                if activeCooldownTimers[character] then
                    if activeCooldownTimers[character].billboard then pcall(function() activeCooldownTimers[character].billboard:Destroy() end) end
                    if activeCooldownTimers[character].connection then activeCooldownTimers[character].connection:Disconnect() end
                    activeCooldownTimers[character] = nil
                end
                if slasherActiveCooldownTimers[character] then
                    for _, data in pairs(slasherActiveCooldownTimers[character]) do
                        if data.billboard then pcall(function() data.billboard:Destroy() end) end
                        if data.connection then data.connection:Disconnect() end
                    end
                    slasherActiveCooldownTimers[character] = nil
                end
                if onexActiveCooldownTimers[character] then
                    for _, data in pairs(onexActiveCooldownTimers[character]) do
                        if data.billboard then pcall(function() data.billboard:Destroy() end) end
                        if data.connection then data.connection:Disconnect() end
                    end
                    onexActiveCooldownTimers[character] = nil
                end
                if sixerActiveCooldownTimers[character] then
                    for _, data in pairs(sixerActiveCooldownTimers[character]) do
                        if data.billboard then pcall(function() data.billboard:Destroy() end) end
                        if data.connection then data.connection:Disconnect() end
                    end

                if noliActiveCooldownTimers[character] then
