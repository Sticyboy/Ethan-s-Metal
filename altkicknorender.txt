-- Pure Player Monitor - KICKS YOU when ANY target user leaves
-- With 3D Rendering Toggle (Press P to toggle)
-- SINGLE EXECUTION PROTECTION - Cannot run twice

-- ============================================
-- PREVENT MULTIPLE EXECUTIONS
-- ============================================

-- Check if script has already been executed
if _G.PlayerMonitorExecuted then
    print("❌ Script already running! Cannot execute twice.")
    return
end

-- Mark as executed immediately
_G.PlayerMonitorExecuted = true

local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer
local RunService = game:GetService("RunService")
local UserInputService = game:GetService("UserInputService")

-- ============================================
-- 3D RENDERING TOGGLE
-- ============================================
local renderingEnabled = true

-- Function to toggle rendering
local function toggleRendering()
    renderingEnabled = not renderingEnabled
    RunService:Set3dRenderingEnabled(renderingEnabled)
    
    if renderingEnabled then
        print("3D Rendering: ON 🟢")
    else
        print("3D Rendering: OFF 🔴")
    end
end

-- Connect P key press (only if not already connected)
if not _G.RenderingToggleConnected then
    _G.RenderingToggleConnected = true
    UserInputService.InputBegan:Connect(function(input, gameProcessed)
        if gameProcessed then return end
        
        if input.KeyCode == Enum.KeyCode.P then
            toggleRendering()
        end
    end)
end

print("Press P to toggle 3D rendering ON/OFF")
print("Current state: ON 🟢")

-- ============================================
-- PLAYER MONITOR - KICK ON LEAVE
-- ============================================

-- Player Monitor Configuration - List of users to watch
local TARGET_USERNAMES = {
    "gwgfdghydvh",
    "vrreblux",
    "scrolladdicttt",
    "guragurababala",
    "mafala8859",
    "lvl2crookboss",
    "flamingfanbloxfruits",
    "cminorpentscale",
    "duxxiitowerdefense",
    "chimeracos"
}

-- Track which users have joined at least once
local usersWhoJoined = {}
local kickExecuted = false

-- Function to check if ANY target user is in server
local function AnyTargetInServer()
    for _, player in ipairs(Players:GetPlayers()) do
        for _, targetName in ipairs(TARGET_USERNAMES) do
            if player.Name:lower() == targetName:lower() then
                return true, player.Name
            end
        end
    end
    return false, nil
end

-- Function to check if a specific user is in server
local function IsUserInServer(username)
    for _, player in ipairs(Players:GetPlayers()) do
        if player.Name:lower() == username:lower() then
            return true
        end
    end
    return false
end

-- Function to kick yourself from the game
local function KickYourself(leavingUser)
    if kickExecuted then return end
    kickExecuted = true
    print("👋 User '" .. leavingUser .. "' left! Kicking self from game...")
    
    -- Method 1: Kick yourself
    pcall(function()
        LocalPlayer:Kick("Target user left the server - Exiting game")
    end)
    
    -- Method 2: Shutdown as backup (if kick doesn't work)
    task.wait(0.5)
    pcall(function()
        game:Shutdown()
    end)
end

-- Initialize user tracking
for _, username in ipairs(TARGET_USERNAMES) do
    usersWhoJoined[username:lower()] = false
end

-- Initial check for any target users already in server
local anyInServer, firstUser = AnyTargetInServer()
if anyInServer then
    print("✅ Users already in server. Monitoring all targets for leave...")
    for _, player in ipairs(Players:GetPlayers()) do
        for _, targetName in ipairs(TARGET_USERNAMES) do
            if player.Name:lower() == targetName:lower() then
                usersWhoJoined[targetName:lower()] = true
                print("  - " .. targetName .. " is in server")
            end
        end
    end
else
    print("⏳ No target users in server. Waiting for any to join first...")
end

-- Monitor for users joining (only if not already monitoring)
if not _G.PlayerAddedConnected then
    _G.PlayerAddedConnected = true
    Players.PlayerAdded:Connect(function(player)
        for _, targetName in ipairs(TARGET_USERNAMES) do
            if player.Name:lower() == targetName:lower() then
                usersWhoJoined[targetName:lower()] = true
                print("✅ User '" .. targetName .. "' joined the server! Will kick self when ANY target leaves.")
                break
            end
        end
    end)
end

-- Monitor for users leaving (only if not already monitoring)
if not _G.PlayerRemovingConnected then
    _G.PlayerRemovingConnected = true
    Players.PlayerRemoving:Connect(function(player)
        for _, targetName in ipairs(TARGET_USERNAMES) do
            if player.Name:lower() == targetName:lower() and usersWhoJoined[targetName:lower()] then
                KickYourself(targetName)
                break
            end
        end
    end)
end

-- Periodic check in case we missed events (only start once)
if not _G.PeriodicCheckStarted then
    _G.PeriodicCheckStarted = true
    task.spawn(function()
        while true do
            task.wait(2)
            
            -- Check if any user who joined is now gone
            for _, targetName in ipairs(TARGET_USERNAMES) do
                local targetLower = targetName:lower()
                if usersWhoJoined[targetLower] and not IsUserInServer(targetName) then
                    KickYourself(targetName)
                end
            end
        end
    end)
end

print("══════════════════════════════════════════════════════════════════")
print("   PURE PLAYER MONITOR - KICK ON TARGET LEAVE")
print("   SINGLE EXECUTION PROTECTION ACTIVE")
print("══════════════════════════════════════════════════════════════════")
print("🔍 Monitoring for " .. #TARGET_USERNAMES .. " users: " .. table.concat(TARGET_USERNAMES, ", "))
print("⚠ Will KICK YOU from the game when ANY target user leaves after having joined")
print("🛡️ Script can only be executed once!")
