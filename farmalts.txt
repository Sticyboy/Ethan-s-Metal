-- Function to check if a character (Survivor or Killer) is level 100 or above
local function isCharacterLevel100(milestoneTable, characterName)
    local milestoneName = characterName .. "Milestone"
    local milestone = milestoneTable:FindFirstChild(milestoneName)
    if milestone then
        return milestone.Value >= 100
    end
    return false
end

-- Function to find the next Survivor that is NOT level 100 based on priority order
local function findNonLevel100Survivor()
    for _, survivorName in ipairs(allSurvivors) do
        if not isCharacterLevel100(survivorsMilestones, survivorName) then
            return survivorName
        end
    end
    return nil -- All survivors are level 100 or above
end

-- Function to find the next Killer that is NOT level 100 based on priority order
local function findNonLevel100Killer()
    for _, killerName in ipairs(allKillers) do
        if not isCharacterLevel100(killersMilestones, killerName) then
            return killerName
        end
    end
    return nil -- All killers are level 100 or above
end

-- Function to INSTANTLY teleport to a survivor
local function TeleportToSurvivor(survivor)
    if not LocalPlayer.Character then return false end
    
    local hrp = LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    local survivorHRP = survivor:FindFirstChild("HumanoidRootPart")
    
    if not hrp or not survivorHRP then return false end
    
    local behindPosition = survivorHRP.Position + (survivorHRP.CFrame.LookVector * -2) + Vector3.new(0, 1, 0)
    hrp.CFrame = CFrame.new(behindPosition, survivorHRP.Position)
    
    return true
end

-- Function to hunt a survivor
local function HuntSurvivor(survivor, killerType)
    if not LocalPlayer.Character then
        print("No character found")
        return false
    end
    
    local hrp = LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    local survivorHRP = survivor:FindFirstChild("HumanoidRootPart")
    local survivorHumanoid = survivor:FindFirstChild("Humanoid")
    
    if not hrp or not survivorHRP or not survivorHumanoid then
        print("Missing required parts")
        return false
    end
    
    print("→ Hunting: " .. survivor.Name .. " (Health: " .. survivorHumanoid.Health .. ") as " .. killerType)
    
    TeleportToSurvivor(survivor)
    task.wait(0.05)
    
    local attackCount = 0
    local startTime = tick()
    
    while survivorHumanoid and survivorHumanoid.Health > 0 do
        attackCount = attackCount + 1
        
        if attackCount % 3 == 0 then
            TeleportToSurvivor(survivor)
        end
        
        Attack(killerType)
        task.wait(STAB_DELAY)
        
        if not survivor.Parent then 
            print("→ Survivor model removed")
            break 
        end
        
        survivorHRP = survivor:FindFirstChild("HumanoidRootPart")
        survivorHumanoid = survivor:FindFirstChild("Humanoid")
        
        if not survivorHRP or not survivorHumanoid then
            print("→ Survivor lost parts")
            break
        end
        
        if attackCount % 5 == 0 then
            print("→ Health: " .. survivorHumanoid.Health .. " (Attacks: " .. attackCount .. ")")
        end
    end
    
    local timeTaken = tick() - startTime
    print("✓ Killed " .. survivor.Name .. " in " .. math.floor(timeTaken * 10) / 10 .. "s with " .. attackCount .. " attacks")
    return true
end

-- SURVIVOR MODE FUNCTIONS
local function GetIncompleteGenerators()
    local generators = {}
    local mapFolder = workspace:FindFirstChild("Map") 
        and workspace.Map:FindFirstChild("Ingame") 
        and workspace.Map.Ingame:FindFirstChild("Map")
    
    if mapFolder then
        for _, obj in ipairs(mapFolder:GetChildren()) do
            if obj.Name == "Generator" and obj:FindFirstChild("Progress") then
                if obj.Progress.Value < 100 then
                    table.insert(generators, obj)
                end
            end
        end
    end
    return generators
end

local function LeaveGenerator(generator)
    if generator and generator.Remotes and generator.Remotes.RF then
        pcall(function()
            generator.Remotes.RF:InvokeServer("Leave")
        end)
    end
end

local function GetGeneratorPosition(generator)
    local positions = generator:FindFirstChild("Positions")
    if positions then
        local center = positions:FindFirstChild("Center")
        if center then
            return center.Position
        end
    end
    return generator.Main and generator.Main.Position or generator:GetPivot().Position
end

local function CompleteGeneratorFast(generator)
    local character = LocalPlayer.Character
    if not character or not character:FindFirstChild("HumanoidRootPart") then
        return false
    end
    
    local hrp = character.HumanoidRootPart
    local genPosition = GetGeneratorPosition(generator)
    
    hrp.CFrame = CFrame.new(genPosition + Vector3.new(0, 1, 0))
    task.wait(0.05)
    
    local prompt = generator.Main and generator.Main:FindFirstChild("Prompt")
    if prompt then
        fireproximityprompt(prompt)
    end
    
    local puzzlesCompleted = 0
    
    while generator.Progress.Value < 100 do
        if generator.Remotes and generator.Remotes.RE then
            generator.Remotes.RE:FireServer()
            puzzlesCompleted = puzzlesCompleted + 1
            
            task.wait(PUZZLE_COMPLETE_DELAY)
            
            if puzzlesCompleted >= PUZZLES_BEFORE_LEAVE and generator.Progress.Value < 100 then
                LeaveGenerator(generator)
                task.wait(LEAVE_DELAY)
                
                hrp.CFrame = CFrame.new(genPosition + Vector3.new(0, 1, 0))
                task.wait(REENTER_DELAY)
                
                if prompt then
                    fireproximityprompt(prompt)
                end
                
                puzzlesCompleted = 0
            end
        end
    end
    
    LeaveGenerator(generator)
    return true
end

-- Main loop function
local function StartLoop()
    print("══════════════════════════════════════════════════════════════════")
    print("    AUTO MODE STARTED - ULTRA FAST")
    print("    Supporting: Noli, 1x1x1x1, Slasher, JohnDoe, Nosferatu, c00lkidd, Sixer")
    print("    Authorized for: " .. LocalPlayer.Name)
    print("    AUTO CHARACTER LEVELING ACTIVE")
    print("══════════════════════════════════════════════════════════════════")
    
    while true do
        local isSurvivor = IsPlayerSurvivor()
        local killerType = GetKillerType()
        
        if isSurvivor then
            print("▶ SURVIVOR MODE (doing generators)")
            
            if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
                local character = LocalPlayer.Character
                local hrp = character.HumanoidRootPart
                local originalPosition = hrp.CFrame
                
                local generators = GetIncompleteGenerators()
                
                if #generators > 0 then
                    print("Found " .. #generators .. " generators")
                    
                    for i, generator in ipairs(generators) do
                        print("  Generator " .. i .. "/" .. #generators)
                        
                        pcall(function()
                            CompleteGeneratorFast(generator)
                        end)
                        
                        if i < #generators then
                            task.wait(BETWEEN_GENERATORS_DELAY)
                        end
                    end
                    
                    hrp.CFrame = originalPosition
                    print("✓ All generators done!")
                end
            end
            
            task.wait(LOOP_DELAY)
            
        elseif killerType then
            print("▶ KILLER MODE - Playing as: " .. killerType)
            
            if LocalPlayer.Character then
                local survivors = GetLivingSurvivors()
                
                if #survivors > 0 then
                    print("Found " .. #survivors .. " survivors")
                    
                    for i, survivor in ipairs(survivors) do
                        print("  Hunting " .. i .. "/" .. #survivors .. ": " .. survivor.Name)
                        
                        local success, err = pcall(function()
                            HuntSurvivor(survivor, killerType)
                        end)
                        
                        if not success then
                            print("  Error:", err)
                        end
                        
                        local newKillerType = GetKillerType()
                        if not newKillerType or newKillerType ~= killerType then
                            print("Killer type changed or no longer killer")
                            break
                        end
                        
                        if i < #survivors then
                            task.wait(BETWEEN_SURVIVORS_DELAY)
                        end
                    end
                else
                    print("No survivors found")
                end
            else
                print("Waiting for character...")
            end
            
            task.wait(LOOP_DELAY)
            
        else
            print("▶ WAITING (not survivor or killer)")
            task.wait(LOOP_DELAY)
        end
    end
end

-- Start the main loop
