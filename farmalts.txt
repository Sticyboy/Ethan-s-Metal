-- Function to switch to a Survivor
local function switchToSurvivor(survivorName)
    local args = {
        "EquipState",
        {
            game:GetService("ReplicatedStorage"):WaitForChild("Assets"):WaitForChild("Survivors"):WaitForChild(survivorName),
            buffer.fromstring("\001\001")
        }
    }
    RemoteEvent:FireServer(unpack(args))
    print("Switched to Survivor: " .. survivorName .. "!")
end

-- Function to switch to a Killer
local function switchToKiller(killerName)

-- Function to check if a character is level 100 or above
local function isCharacterLevel100(milestoneTable, characterName)
    local milestoneName = characterName .. "Milestone"
    local milestone = milestoneTable:FindFirstChild(milestoneName)
    if milestone then
        return milestone.Value >= 100
    end
    return false
end

-- Function to find the next Survivor not at level 100
local function findNonLevel100Survivor()
    for _, survivorName in ipairs(survivorsOrder) do
        if not isCharacterLevel100(survivorsMilestones, survivorName) then
            return survivorName
        end
    end
    return nil
end

-- Function to find the next Killer not at level 100
local function findNonLevel100Killer()
    for _, killerName in ipairs(killersOrder) do
        if not isCharacterLevel100(killersMilestones, killerName) then
            return killerName
        end
    end
    return nil
end

-- Function to handle Survivor switching
local function switchToNonLevel100Survivor()
    local targetSurvivor = findNonLevel100Survivor()
    
    if targetSurvivor then
        print("Found Survivor not at level 100: " .. targetSurvivor)
        switchToSurvivor(targetSurvivor)
    else

    end
end

-- Function to handle Killer switching

-- Start the character leveling monitor
local function startLevelingMonitor()
    -- Initial switches
    switchToNonLevel100Survivor()
    switchToNonLevel100Killer()
    
    -- Monitor for level ups
    for _, survivorName in ipairs(survivorsOrder) do
        local milestoneName = survivorName .. "Milestone"
        local milestone = survivorsMilestones:FindFirstChild(milestoneName)
        if milestone then
            milestone:GetPropertyChangedSignal("Value"):Connect(function()
                if milestone.Value >= 100 then
                    print(survivorName .. " (Survivor) reached level " .. milestone.Value .. "!")
                    switchToNonLevel100Survivor()
                end
            end)
        end
    end
    
    for _, killerName in ipairs(killersOrder) do
        local milestoneName = killerName .. "Milestone"
        local milestone = killersMilestones:FindFirstChild(milestoneName)
        if milestone then
            milestone:GetPropertyChangedSignal("Value"):Connect(function()
                if milestone.Value >= 100 then
                    print(killerName .. " (Killer) reached level " .. milestone.Value .. "!")
                    switchToNonLevel100Killer()
                end
            end)
        end
    end
    
    print("Monitoring all Survivors and Killers for level 100...")
    print("Survivor priority order: " .. table.concat(survivorsOrder, " → "))
end

-- Configuration - GENERATOR MODE
local PUZZLE_DELAY = 2.8 -- seconds between each puzzle
local BETWEEN_GENERATORS_DELAY = 2 -- seconds between different generators
local LOOP_DELAY = 0.5

-- Configuration - KILLER MODE (Left Click Spam)
local CLICK_DELAY = 0.15
local BETWEEN_SURVIVORS_DELAY = 0.2

-- Function to simulate left click
local function LeftClick()
    pcall(function()
        VirtualInput:SendMouseButtonEvent(0, 0, 0, true, game, 0)
        task.wait(0.02)
        VirtualInput:SendMouseButtonEvent(0, 0, 0, false, game, 0)
        
        task.wait(0.02)
        UserInputService:SendMouseButtonEvent(Enum.UserInputType.MouseButton1, 0, 0, true)
        task.wait(0.02)
        UserInputService:SendMouseButtonEvent(Enum.UserInputType.MouseButton1, 0, 0, false)
    end)
end

-- Function to check if local player is a survivor
local function IsPlayerSurvivor()
    local survivors = workspace:FindFirstChild("Players") and workspace.Players:FindFirstChild("Survivors")
    if not survivors then return false end
    
    for _, survivor in ipairs(survivors:GetChildren()) do
        if survivor:GetAttribute("Username") == LocalPlayer.Name then
            return true
        end
        if survivor.Name == LocalPlayer.Name then
            return true
        end
    end
    return false
end

-- Function to check if local player is a killer
local function IsPlayerKiller()
    local killers = workspace:FindFirstChild("Players") and workspace.Players:FindFirstChild("Killers")
    if not killers then return false end
    
    for _, killer in ipairs(killers:GetChildren()) do
        local isMe = killer:GetAttribute("Username") == LocalPlayer.Name or killer.Name == LocalPlayer.Name
        if isMe then
            return true
        end
    end
    return false
end

-- Function to get killer name
local function GetKillerName()
    local killers = workspace:FindFirstChild("Players") and workspace.Players:FindFirstChild("Killers")
    if not killers then return nil end
    
    for _, killer in ipairs(killers:GetChildren()) do
        local isMe = killer:GetAttribute("Username") == LocalPlayer.Name or killer.Name == LocalPlayer.Name
        if isMe then
            return killer.Name
        end
    end
    return nil
end

-- Function to get all living survivors
local function GetLivingSurvivors()
    local survivors = {}
    local survivorsFolder = workspace:FindFirstChild("Players") and workspace.Players:FindFirstChild("Survivors")
    
    if survivorsFolder then
        for _, survivor in ipairs(survivorsFolder:GetChildren()) do
            local humanoid = survivor:FindFirstChild("Humanoid")
            if humanoid and humanoid.Health > 0 then
                table.insert(survivors, survivor)
            end
        end
    end
    return survivors
end

-- Function to get the Map folder
local function GetMapFolder()
    return workspace:FindFirstChild("Map") 
        and workspace.Map:FindFirstChild("Ingame") 
        and workspace.Map.Ingame:FindFirstChild("Map")
end

-- Function to get all incomplete generators
local function GetIncompleteGenerators()
    local generators = {}
    local mapFolder = GetMapFolder()
    
    if mapFolder then
        for _, obj in ipairs(mapFolder:GetChildren()) do
            if obj.Name == "Generator" and obj:FindFirstChild("Progress") then
                if obj.Progress.Value < 100 then
                    table.insert(generators, obj)
                end
            end
        end
    end
    return generators
end

-- Function to get generator position
local function GetGeneratorPosition(generator)
    local positions = generator:FindFirstChild("Positions")
    if positions then
        local center = positions:FindFirstChild("Center")
        if center then
            return center.Position
        end
    end
    if generator:FindFirstChild("Main") then
        return generator.Main.Position
    end
    return generator:GetPivot().Position
end

-- Function to fire proximity prompt
local function FireProximityPrompt(generator)
    if generator and generator:FindFirstChild("Main") then
        local prompt = generator.Main:FindFirstChild("Prompt")
        if prompt then
            pcall(function()
                fireproximityprompt(prompt)
                print("  ✓ Fired proximity prompt")
                return true
            end)
        else
            print("  ✗ No Prompt found")
        end
    end
    return false
end

-- Function to complete one puzzle
local function CompletePuzzle(generator)
    if generator and generator.Remotes and generator.Remotes.RE then
        pcall(function()
            generator.Remotes.RE:FireServer()
        end)
        return true
    end
    return false
end

-- Function to complete a generator
local function CompleteGenerator(generator)
    local character = LocalPlayer.Character
    if not character or not character:FindFirstChild("HumanoidRootPart") then
        print("  No character found")
        return false
    end
    
    local hrp = character.HumanoidRootPart
    local genPosition = GetGeneratorPosition(generator)
    
    -- Teleport to generator
    hrp.CFrame = CFrame.new(genPosition + Vector3.new(0, 1, 0))
    task.wait(0.2)
    
    -- Fire proximity prompt to enter
    FireProximityPrompt(generator)
    task.wait(0.5)
    
    -- Keep completing puzzles every 2.8 seconds until generator is 100%
    local puzzlesCompleted = 0
    local startProgress = generator.Progress.Value
    
    print("  Starting generator at " .. startProgress .. "%")
    print("  Will complete a puzzle every " .. PUZZLE_DELAY .. " seconds")
    
    while generator.Progress.Value < 100 do
        CompletePuzzle(generator)
        puzzlesCompleted = puzzlesCompleted + 1
        print("  → Puzzle #" .. puzzlesCompleted .. " | Progress: " .. generator.Progress.Value .. "%")
        
        if generator.Progress.Value < 100 then
            task.wait(PUZZLE_DELAY)
        end
        
        if not generator.Parent then
            print("  Generator was destroyed")
            break
        end
    end
    
    print("  ✓ Generator completed! (" .. puzzlesCompleted .. " puzzles, " .. generator.Progress.Value .. "%)")
    return true
end

-- Function to INSTANTLY teleport to a survivor
local function TeleportToSurvivor(survivor)
    if not LocalPlayer.Character then return false end
    
    local hrp = LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    local survivorHRP = survivor:FindFirstChild("HumanoidRootPart")
    
    if not hrp or not survivorHRP then return false end
    
    local behindPosition = survivorHRP.Position + (survivorHRP.CFrame.LookVector * -2) + Vector3.new(0, 1, 0)
    hrp.CFrame = CFrame.new(behindPosition, survivorHRP.Position)
    
    return true
end

-- Function to hunt a survivor
local function HuntSurvivor(survivor, killerName)
    if not LocalPlayer.Character then
        print("No character found")
        return false
    end
    
    local hrp = LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    local survivorHRP = survivor:FindFirstChild("HumanoidRootPart")
    local survivorHumanoid = survivor:FindFirstChild("Humanoid")
    
    if not hrp or not survivorHRP or not survivorHumanoid then
        print("Missing required parts")
        return false
    end
    
    print("→ Hunting: " .. survivor.Name .. " (Health: " .. survivorHumanoid.Health .. ") as " .. killerName)
    
    TeleportToSurvivor(survivor)
    task.wait(0.05)
    
    local clickCount = 0
    local startTime = tick()
    local maxTime = 30
    
    while survivorHumanoid and survivorHumanoid.Health > 0 and tick() - startTime < maxTime do
        clickCount = clickCount + 1
        
        if clickCount % 5 == 0 then
            TeleportToSurvivor(survivor)
        end
        
        LeftClick()
        task.wait(CLICK_DELAY)
        
        if not survivor.Parent then 
            print("→ Survivor model removed")
            break 
        end
        
        survivorHRP = survivor:FindFirstChild("HumanoidRootPart")
        survivorHumanoid = survivor:FindFirstChild("Humanoid")
        
        if not survivorHRP or not survivorHumanoid then
            print("→ Survivor lost parts")
            break
        end
        
        if clickCount % 10 == 0 then
            print("→ Health: " .. survivorHumanoid.Health .. " (Clicks: " .. clickCount .. ")")
        end
    end
    
    local timeTaken = tick() - startTime
    if survivorHumanoid and survivorHumanoid.Health > 0 then
        print("⚠ Timeout hunting " .. survivor.Name)
    else
        print("✓ Killed " .. survivor.Name .. " in " .. math.floor(timeTaken * 10) / 10 .. "s with " .. clickCount .. " clicks")
    end
    return true
end

-- Main loop function
local function StartLoop()
    print("══════════════════════════════════════════════════════════════════")
    print("    AUTO MODE STARTED - ULTRA FAST")
    print("    Authorized for: " .. LocalPlayer.Name)
    print("    Survivor: Auto-completes generators (puzzle every " .. PUZZLE_DELAY .. "s)")
    print("    Killer: Auto-hunts survivors (LEFT CLICK SPAM)")
    print("    Auto Character Leveling: ACTIVE")
    print("══════════════════════════════════════════════════════════════════")
    
    while true do
        local isSurvivor = IsPlayerSurvivor()
        local isKiller = IsPlayerKiller()
        local killerName = GetKillerName()
        
        if isSurvivor then
            print("▶ SURVIVOR MODE (doing generators)")
            
            if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
                local character = LocalPlayer.Character
                local hrp = character.HumanoidRootPart
                local originalPosition = hrp.CFrame
                
                local generators = GetIncompleteGenerators()
                
                if #generators > 0 then
                    print("Found " .. #generators .. " generators to complete")
                    
                    for i, generator in ipairs(generators) do
                        print("\n  Generator " .. i .. "/" .. #generators .. " (Progress: " .. generator.Progress.Value .. "%)")
                        
                        pcall(function()
                            CompleteGenerator(generator)
                        end)
                        
                        if i < #generators then
                            print("  Waiting " .. BETWEEN_GENERATORS_DELAY .. " seconds before next generator...")
                            task.wait(BETWEEN_GENERATORS_DELAY)
                        end
                    end
                    
                    if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
                        hrp.CFrame = originalPosition
                    end
                    print("\n✓ All generators done!")
                else
                    print("No incomplete generators found")
                end
            else
                print("Waiting for character...")
            end
            
            task.wait(LOOP_DELAY)
            
        elseif isKiller then
            print("▶ KILLER MODE - Playing as: " .. (killerName or "Unknown") .. " (LEFT CLICK SPAM)")
            
            if LocalPlayer.Character then
                local survivors = GetLivingSurvivors()
                
                if #survivors > 0 then
                    print("Found " .. #survivors .. " survivors")
                    
                    for i, survivor in ipairs(survivors) do
                        print("  Hunting " .. i .. "/" .. #survivors .. ": " .. survivor.Name)
                        
                        local success, err = pcall(function()
                            HuntSurvivor(survivor, killerName or "Unknown")
                        end)
                        
                        if not success then
                            print("  Error:", err)
                        end
                        
                        if not IsPlayerKiller() then
                            print("No longer killer")
                            break
                        end
                        
                        if i < #survivors then
                            task.wait(BETWEEN_SURVIVORS_DELAY)
                        end
                    end
                else
                    print("No survivors found")
                end
            else
                print("Waiting for character...")
            end
            
            task.wait(LOOP_DELAY)
            
        else
            print("▶ WAITING (not survivor or killer)")
            task.wait(LOOP_DELAY)
        end
    end
end

-- Start the main loop
local success, err = pcall(StartLoop)
if not success then
    print("Loop failed:", err)
end
