--// Stamina System GUI (Skateboard transparency + Trick audio detection)

local UserInputService = game:GetService("UserInputService")
local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local Workspace = game:GetService("Workspace")

-- ScreenGui Setup
local ScreenGui = Instance.new("ScreenGui")
ScreenGui.Name = "StaminaGui"
ScreenGui.ZIndexBehavior = Enum.ZIndexBehavior.Global
ScreenGui.Parent = game:GetService("CoreGui")

-- Stamina Label
local StaminaLabel = Instance.new("TextLabel")
StaminaLabel.Parent = ScreenGui
StaminaLabel.Size = UDim2.new(0, 350, 0, 30)
StaminaLabel.Position = UDim2.new(0, 20, 1, -50)
StaminaLabel.BackgroundTransparency = 0.3	
StaminaLabel.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
StaminaLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
StaminaLabel.TextScaled = true
StaminaLabel.Font = Enum.Font.SourceSansBold
StaminaLabel.Text = "Stamina: 100"

-- Buttons
local SurvivorButton = Instance.new("TextButton")
SurvivorButton.Parent = ScreenGui
SurvivorButton.Size = UDim2.new(0, 80, 0, 30)
SurvivorButton.Position = UDim2.new(0, 380, 1, -50)
SurvivorButton.Text = "Survivor"
SurvivorButton.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
SurvivorButton.TextColor3 = Color3.new(1,1,1)

local KillerButton = Instance.new("TextButton")
KillerButton.Parent = ScreenGui
KillerButton.Size = UDim2.new(0, 80, 0, 30)
KillerButton.Position = UDim2.new(0, 470, 1, -50)
KillerButton.Text = "Killer"
KillerButton.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
KillerButton.TextColor3 = Color3.new(1,1,1)

-- Vars
local currentMode = "Survivor"
local maxStamina = 100
local stamina = 100
local drainRate = 10
local regenRate = 20
local SKATING_DRAIN_MULTIPLIER = 1.35  -- 1.35x drain when skating

-- Skating state (based on skateboard transparency)
local isSkating = false
local skateboardPart = nil
local veeronicaChar = nil

-- Trick audio ID
local TRICK_AUDIO_ID = "rbxassetid://91937289286100"
local trickSound = nil

-- Module variables
local timeUntilStaminaRecovers = 0
local staminaPenaltyActive = false
local holdingShift = false
local isSprinting = false
local wasSprinting = false

local localPlayer = Players.LocalPlayer
local localCharacter = localPlayer.Character or localPlayer.CharacterAdded:Wait()
local humanoid = localCharacter:WaitForChild("Humanoid")
local humanoidRootPart = localCharacter:WaitForChild("HumanoidRootPart")

local isMoving = false
local VELOCITY_THRESHOLD = 0.4

local KILLER_CHECK_RANGE = 110
local isInKillerRange = false

-- Function to get the skateboard directly from the path
local function getSkateboard()
    -- Try direct path: workspace.Players.Survivors.Veeronica
    local survivors = Workspace:FindFirstChild("Players") and 
                     Workspace.Players:FindFirstChild("Survivors")
    
    if not survivors then return nil end
    
    -- Look for Veeronica in Survivors
    veeronicaChar = survivors:FindFirstChild("Veeronica")
    if not veeronicaChar then return nil end
    
    -- Check if this Veeronica is you (Username attribute with capital U)
    if veeronicaChar:GetAttribute("Username") == localPlayer.Name then
        -- Find the Skateboard
        local skateboard = veeronicaChar:FindFirstChild("Skateboard")
        return skateboard
    end
    
    return nil
end

-- Function to detect trick audio in HumanoidRootPart
local function setupTrickAudioDetection()
    -- Look for your Veeronica character
    local survivors = Workspace:FindFirstChild("Players") and 
                     Workspace.Players:FindFirstChild("Survivors")
    
    if not survivors then return end
    
    veeronicaChar = survivors:FindFirstChild("Veeronica")
    if not veeronicaChar then return end
    
    -- Check if this Veeronica is you
    if veeronicaChar:GetAttribute("Username") ~= localPlayer.Name then return end
    
    -- Look in HumanoidRootPart for the sound
    local humanoidRootPart = veeronicaChar:FindFirstChild("HumanoidRootPart")
    if not humanoidRootPart then return end
    
    -- Find the trick sound
    for _, child in ipairs(humanoidRootPart:GetChildren()) do
        if child:IsA("Sound") and child.SoundId == TRICK_AUDIO_ID then
            trickSound = child
            break
        end
    end
    
    if not trickSound then
        -- If not found, wait for it to be added
        humanoidRootPart.ChildAdded:Connect(function(child)
            if child:IsA("Sound") and child.SoundId == TRICK_AUDIO_ID then
                trickSound = child
                -- Connect to the sound playing
                trickSound.Played:Connect(function()
                    -- Give 4 stamina immediately
                    stamina = math.min(stamina + 4, maxStamina)
                    print("🎮 +4 stamina gained from trick! (Total:", stamina, ")")
                end)
            end
        end)
        return
    end
    
    -- Connect to the sound playing
    trickSound.Played:Connect(function()
        -- Give 4 stamina immediately
        stamina = math.min(stamina + 4, maxStamina)
        print("🎮 +4 stamina gained from trick! (Total:", stamina, ")")
    end)
end

-- Update character references when character respawns
localPlayer.CharacterAdded:Connect(function(character)
    localCharacter = character
    humanoid = character:WaitForChild("Humanoid")
    humanoidRootPart = character:WaitForChild("HumanoidRootPart")
    
    stamina = maxStamina
    timeUntilStaminaRecovers = 0
    staminaPenaltyActive = false
    isSprinting = false
    isSkating = false
    
    -- Reset references
    skateboardPart = nil
    veeronicaChar = nil
    trickSound = nil
end)

-- Function to check if player is moving
local function updateMovementCheck()
    if humanoidRootPart and humanoidRootPart:IsDescendantOf(workspace) then
        isMoving = humanoidRootPart.Velocity.Magnitude > VELOCITY_THRESHOLD
    end
end

-- Function to check if killer is near other players
local function updateKillerRangeCheck()
    if not localCharacter or not humanoidRootPart then 
        isInKillerRange = false
        return 
    end
    
    local killerPosition = humanoidRootPart.Position
    local foundPlayer = false
    
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= localPlayer and player.Character then
            local targetCharacter = player.Character
            local targetRootPart = targetCharacter:FindFirstChild("HumanoidRootPart")
            
            if targetRootPart then
                local distance = (killerPosition - targetRootPart.Position).Magnitude
                if distance <= KILLER_CHECK_RANGE then
                    foundPlayer = true
                    break
                end
            end
        end
    end
    
    isInKillerRange = foundPlayer
end

-- Function to check if we can regenerate stamina
local function canRegenStamina()
    return timeUntilStaminaRecovers <= 0 and not staminaPenaltyActive
end

-- Switch Modes
local function setMode(mode)
    if mode == "Survivor" then
        maxStamina = 100
        stamina = maxStamina
        drainRate = 10
        regenRate = 20
        SurvivorButton.BackgroundColor3 = Color3.fromRGB(0, 100, 0)
        KillerButton.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
    elseif mode == "Killer" then
        maxStamina = 110
        stamina = maxStamina
        drainRate = 9.5
        regenRate = 21
        SurvivorButton.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
        KillerButton.BackgroundColor3 = Color3.fromRGB(100, 0, 0)
    end
    currentMode = mode
    isInKillerRange = false
    timeUntilStaminaRecovers = 0
    staminaPenaltyActive = false
    isSprinting = false
    wasSprinting = false
    isSkating = false
end

SurvivorButton.MouseButton1Click:Connect(function()
    setMode("Survivor")
end)

KillerButton.MouseButton1Click:Connect(function()
    setMode("Killer")
end)

-- Input events
UserInputService.InputBegan:Connect(function(input, gp)
    if input.KeyCode == Enum.KeyCode.LeftShift then
        holdingShift = true
    end
end)

UserInputService.InputEnded:Connect(function(input, gp)
    if input.KeyCode == Enum.KeyCode.LeftShift then
        holdingShift = false
    end
end)

-- Monitor skateboard transparency changes
local function updateSkateboardState()
    if not skateboardPart or not skateboardPart:IsDescendantOf(Workspace) then
        skateboardPart = getSkateboard()
    end
    
    if skateboardPart then
        isSkating = (skateboardPart.Transparency == 0)
    end
end

-- Initial setup
updateSkateboardState()
setupTrickAudioDetection()

-- Heartbeat loop (runs every frame for fast detection)
RunService.Heartbeat:Connect(function(dt)
    -- Update checks
    updateMovementCheck()
    updateSkateboardState()  -- Check skateboard every frame
    
    if currentMode == "Killer" then
        updateKillerRangeCheck()
    end
    
    -- Determine if we should be draining stamina
    local shouldDrain = false
    
    if isSkating then
        -- When skating, drain stamina while moving (shift state doesn't matter)
        shouldDrain = isMoving
    elseif currentMode == "Survivor" then
        -- Normal sprinting requires shift + moving
        shouldDrain = holdingShift and isMoving and stamina > 0
    elseif currentMode == "Killer" then
        -- Killer sprinting requires shift + moving + in range
        shouldDrain = holdingShift and isMoving and isInKillerRange and stamina > 0
    end
    
    -- Track sprinting state changes (for UI only)
    wasSprinting = isSprinting
    isSprinting = shouldDrain
    
    -- Handle sprinting ended (applies to both normal sprinting and skating)
    if wasSprinting and not isSprinting then
        if timeUntilStaminaRecovers > 0.1 then
            timeUntilStaminaRecovers = math.clamp(timeUntilStaminaRecovers + 0.1, 0, 3)
        else
            timeUntilStaminaRecovers = 0.1
        end
    end
    
    -- Handle stamina changes
    if shouldDrain and stamina > 0 then
        -- Calculate drain rate (1.35x for skating)
        local currentDrainRate = drainRate
        if isSkating then
            currentDrainRate = drainRate * SKATING_DRAIN_MULTIPLIER  -- 1.35x drain when skating
        end
        stamina = stamina - currentDrainRate * dt
        stamina = math.max(stamina, 0)
        
        -- Increase recovery timer while draining (for both skating and sprinting)
        timeUntilStaminaRecovers = math.clamp(timeUntilStaminaRecovers + dt * 0.05, 0.2, 3)
        
        if stamina <= 0 then
            stamina = 0
            isSprinting = false
            timeUntilStaminaRecovers = 2
            staminaPenaltyActive = true
            localCharacter:SetAttribute("StaminaPenaltyActive", true)
        end
    else
        -- Not draining, decrement recovery timer
        if timeUntilStaminaRecovers > 0 then
            timeUntilStaminaRecovers = math.clamp(timeUntilStaminaRecovers - dt, 0, 3)
        end
        
        -- Clear penalty when timer runs out
        if timeUntilStaminaRecovers <= 0 and staminaPenaltyActive then
            staminaPenaltyActive = false
            localCharacter:SetAttribute("StaminaPenaltyActive", nil)
        end
        
        -- Regenerate stamina if allowed (not while skating)
        if not isSkating and canRegenStamina() and stamina < maxStamina then
            stamina = stamina + regenRate * dt
            stamina = math.min(stamina, maxStamina)
        end
    end
    
    -- Update label
    local statusText = ""
    
    -- Add skating status
    if isSkating then
        statusText = statusText .. " (SKATING)"
    end
    
    if currentMode == "Killer" then
        if isInKillerRange then
            statusText = statusText .. " (IN RANGE)"
        else
            statusText = statusText .. " (OUT OF RANGE)"
        end
    end
    
    if not isMoving then
        statusText = statusText .. " (STATIONARY)"
    end
    
    if shouldDrain then
        if stamina > 0 then
            if isSkating then
                statusText = statusText .. " (SKATING)"
            else
                statusText = statusText .. " (SPRINTING)"
            end
        else
            statusText = statusText .. " (EXHAUSTED)"
        end
    end
    
    if staminaPenaltyActive then
        statusText = statusText .. " (PENALTY)"
    end
    
    if stamina < maxStamina and not shouldDrain then
        if timeUntilStaminaRecovers > 0 then
            statusText = statusText .. string.format(" (DELAY %.1fs)", timeUntilStaminaRecovers)
        elseif not staminaPenaltyActive and not isSkating then
            statusText = statusText .. " (REGENERATING)"
        end
    elseif stamina >= maxStamina then
        statusText = statusText .. " (FULL)"
    end
    
    StaminaLabel.Text = string.format("Stamina: %d%s", math.floor(stamina), statusText)
    
    -- Color coding
    if stamina < 20 then
        StaminaLabel.TextColor3 = Color3.fromRGB(255, 50, 50)
        StaminaLabel.BackgroundColor3 = Color3.fromRGB(50, 0, 0)
    elseif stamina < 50 then
        StaminaLabel.TextColor3 = Color3.fromRGB(255, 200, 50)
        StaminaLabel.BackgroundColor3 = Color3.fromRGB(30, 15, 0)
    else
        StaminaLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
        StaminaLabel.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
    end
    
    -- Update button transparency
    SurvivorButton.BackgroundTransparency = currentMode == "Survivor" and 0.1 or 0.3
    KillerButton.BackgroundTransparency = currentMode == "Killer" and 0.1 or 0.3
end)

-- Initial setup
setMode("Survivor")
